```python
from groq import Groq
import os

def summarize_paragraph(paragraph):
    """Summarize into 1-2 sentences."""
    # Step 1: Validate input
    if not isinstance(paragraph, str) or len(paragraph) == 0:
        raise ValueError("paragraph must be non-empty string")
    
    # Step 2: Get API key from environment
    api_key = os.getenv("GROQ_API_KEY")
    if not api_key:
        raise ValueError("GROQ_API_KEY not set")
    
    # Step 3: Create Groq client - PASS ONLY api_key
    groq_client = Groq(api_key=api_key)
    
    # Step 4: Create user message for LLM
    user_message = f"Summarize this in 1-2 sentences: {paragraph}"
    
    # Step 5: Call LLM - use model openai/gpt-oss-120b
    llm_response = groq_client.chat.completions.create(
        model="openai/gpt-oss-120b",
        messages=[{"role": "user", "content": user_message}],
        temperature=0.5,
        max_tokens=150
    )
    
    # Step 6: Extract summary from response
    summary_text = llm_response.choices[0].message.content
    word_count = len(summary_text.split())
    
    # Step 7: Return dictionary with EXACT output keys
    return_dict = {
        "summary": summary_text,
        "word_count": word_count
    }
    return return_dict
```

CRITICAL RULES:
- Groq() takes ONLY api_key parameter
- Function parameters go in def function_name(param): NOT in Groq()
- Always use model="openai/gpt-oss-120b"
- Always return a dict with the exact output keys
